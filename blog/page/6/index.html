.
├── adhoc/
│   ├── rolling-reboot.yml
│   └── scripts/
├── README.md

    │   └── zimbra/

    │       ├── hosts
    │       ├── tasks/
    │       └── ...
    ├── group_vars/

    │   └── zimbra-prod

    ├── host_vars/

    └── zimbra-playbook.yml
